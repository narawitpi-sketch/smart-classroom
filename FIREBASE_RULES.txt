rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
  
    // Helper function to check for admin role
    // UPDATE THIS LIST WITH YOUR ADMIN EMAILS
    function isAdmin() {
      return request.auth.token.email == 'narawit.pi@nsru.ac.th' || 
             exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }

    // A better way is to use custom claims, but for now, we'll use a simple admin collection.
    // To make a user an admin, create a document in the 'admins' collection with the user's UID as the document ID.
    // e.g., /admins/{user_uid}
    match /admins/{userId} {
      allow read, write: if false; // Should be managed from the Firebase console or a trusted server environment
    }
    
    // Public data, like issues and rooms
    match /artifacts/{appId}/public/data/{collection}/{docId} {
      allow read, write: if false; // Default deny
    }

    // Issues can be created by ANYONE (Public).
    // Only admins can read all issues or modify them.
    match /artifacts/{appId}/public/data/issues/{issueId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }
    
    // Rooms can be read by ANYONE (so reporter screen can list them).
    match /artifacts/{appId}/public/data/rooms/{roomId} {
      allow read, list: if true;
      allow create, delete: if isAdmin();
    }
    
    // Feedbacks can be created by ANYONE.
    match /artifacts/{appId}/public/data/feedbacks/{feedbackId} {
       allow create: if true;
       allow read: if isAdmin();
    }

    // Inventory can be read by authenticated users (for selection), but modified only by admins.
    match /artifacts/{appId}/public/data/inventory/{itemId} {
       allow read: if request.auth != null;
       allow write: if isAdmin();
    }
  }
}
